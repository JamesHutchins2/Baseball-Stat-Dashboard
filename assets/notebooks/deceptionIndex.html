Pitcher Deception Index

The following aims to develop a statistic using easily accessible data. This statistic aims to quantify the similatrity in pitch execution profiles in comparison to the resulting pitches delivered. The plan is to:

  1. Extract required data.
  2. Train a standard logistic model on a given number of pitches, while accounting for the set of pitches the given pitcher uses.
  3. Resulting socres based on that normalization and model accuracy will provide a "pithcer deception index"

def compute_medians(df: pd.DataFrame) -> pd.DataFrame:
    """
    Compute medians of release variables grouped by pitch_type.
    """
    cols = ['release_pos_x','release_pos_z','release_pos_y',
            'release_spin_rate','release_extension']
    return df.groupby('pitch_type')[cols].median().reset_index()

def compute_variances(df: pd.DataFrame) -> pd.DataFrame:
    """
    Compute variances of release variables grouped by pitch_type.
    """
    cols = ['release_pos_x','release_pos_z','release_pos_y',
            'release_spin_rate','release_extension']
    return df.groupby('pitch_type')[cols].var().reset_index()

def compute_summary(df: pd.DataFrame) -> pd.DataFrame:
    """
    Combine medians and variances into one summary table.
    """
    med = compute_medians(df)
    var = compute_variances(df)
    
    # merge on pitch_type
    summary = pd.merge(med, var, on="pitch_type", suffixes=("_median", "_var"))
    return summary

Implementation

Now let's use a Random Forest classifier to compute deception score.

We will implement cross-validation to get a robust estimate of accuracy, and ensure that our estimate will likely generalize to new data.

To account for a varying number of pitches thrown by a given pitcher, we normalize using the following set pf formulas:

Defining our variables as:

  • Features: $X \in \mathbb{R}^{n \times d}$
  • Labels: $y \in \{1,2,\dots, C\}^n$ (with $C$ pitch types)

Using the corss validation accuracy score:

$$ \text{acc} = \frac{1}{K} \sum_{k=1}^{K} \frac{1}{|D_k|} \sum_{(x_i,y_i) \in D_k} \mathbf{1}\{f_{-k}(x_i) = y_i\} $$

where:

  • $K = 5$ folds
  • $D_k$ = held-out fold
  • $f_{-k}$ = classifier trained without fold $k$

Calcualte a baseline model accuracy

$$ \text{baseline} = \frac{1}{C} $$

Normalize it on the size of the set of possible pitches

$$ \text{norm\_acc} = \frac{\text{acc} - \text{baseline}}{1 - \text{baseline}} $$

Calculate our final Deception score

$$ \text{deception} = 100 \times \Big(1 - \text{norm\_acc}\Big) $$

In[9]:
from sklearn.model_selection import cross_val_score
from sklearn.ensemble import RandomForestClassifier
import numpy as np


def deception_score(df):
    cols = ['release_pos_x','release_pos_z','release_pos_y',
            'release_spin_rate','release_extension']
    
    X = df[cols]
    y = df['pitch_type']
    
    n_classes = y.nunique()
    if n_classes < 2:
        return np.nan  # can't compute with only one pitch type
    
    # let's use a relatively large forest to get a stable estimate
    clf = RandomForestClassifier(n_estimators=200, random_state=42)
    acc = cross_val_score(clf, X, y, cv=5, scoring="accuracy").mean()
    
    # normalize relative to random baseline
    baseline = 1 / n_classes
    norm_acc = (acc - baseline) / (1 - baseline)
    
    deception = 100 * (1 - norm_acc)
    return deception


score = deception_score(df)

from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import StratifiedKFold, cross_val_score
from collections import Counter

FEATURE_COLS_DEFAULT = [
    'release_pos_x','release_pos_z','release_pos_y',
    'release_spin_rate','release_extension'
]
TARGET_COL = 'pitch_type'

# ---------- utilities ----------
def _as_feature_list(feature_cols):
    try:
        return [str(c) for c in (FEATURE_COLS_DEFAULT if feature_cols is None else list(feature_cols))]
    except Exception:
        raise TypeError(f"feature_cols must be list-like of names, got {type(feature_cols)}")

def _dedupe_cols(df: pd.DataFrame) -> pd.DataFrame:
    """Drop duplicate-named columns, keep the first occurrence."""
    if df.columns.duplicated().any():
        # optional: print a quick diagnostic
        dupes = [c for c, n in Counter(df.columns).items() if n > 1]
        print(f"[WARN] Dropping duplicate columns (keeping first): {dupes}")
    return df.loc[:, ~df.columns.duplicated(keep="first")].copy()

# ---------- PDI ----------
def deception_score(df, feature_cols=None):
    feat = _as_feature_list(feature_cols)
    df = _dedupe_cols(df)
    present = [c for c in feat if c in df.columns]
    if not present or TARGET_COL not in df.columns:
        return np.nan
    X = df[present].apply(pd.to_numeric, errors="coerce")
    y = df[TARGET_COL]
    mask = X.notnull().all(axis=1) & y.notnull()
    X, y = X.loc[mask].values, y.loc[mask]
    n_classes = y.nunique()
    if n_classes < 2 or len(y) < 5:
        return np.nan
    clf = RandomForestClassifier(n_estimators=200, random_state=42)
    acc = cross_val_score(clf, X, y, cv=5, scoring="accuracy").mean()
    baseline = 1.0 / n_classes
    norm_acc = (acc - baseline) / (1 - baseline)
    return 100 * (1 - norm_acc)

def build_pooled_dataset(player_ids, year, min_rows=50, feature_cols=None):
    feat = _as_feature_list(feature_cols)
    frames, meta = [], []
    for pid in player_ids:

            continue

        if not present or TARGET_COL not in df.columns:
            continue
        tmp = df[present + [TARGET_COL]].copy()
        tmp[present] = tmp[present].apply(pd.to_numeric, errors="coerce")
        mask = tmp[present].notnull().all(axis=1) & tmp[TARGET_COL].notnull()
        tmp = tmp.loc[mask]

        tmp["pitcher_id"] = pid
        frames.append(tmp)

        raise ValueError("No eligible pitcher data after filtering.")
    pooled = pd.concat(frames, ignore_index=True)
    counts = pd.DataFrame(meta, columns=["pitcher_id","n_rows"])
    return pooled, counts

# ---------- PCA ----------
def fit_global_pca_safe(pooled_df, feature_cols=None, n_components=None, random_state=42, verbose=True):
    feat = _as_feature_list(feature_cols)
    pooled_df = _dedupe_cols(pooled_df)
    present = [c for c in feat if c in pooled_df.columns]
    if not present:
        raise ValueError("None of the requested feature columns are present in pooled_df.")
    X = pooled_df[present].apply(pd.to_numeric, errors="coerce").dropna(axis=0, how="any")

    # drop zero-variance columns
    var = X.var(ddof=0)
    keep = var[var > 0].index.tolist()
    if len(keep) < 2:
        raise ValueError(f"Need 2 numeric, non-constant features. Got: {keep}")
    X = X[keep]
    feature_names = keep

    # clamp PCs
    max_pc = len(feature_names)
    if n_components is None or n_components > max_pc:
        n_components = max_pc

    scaler = StandardScaler().fit(X.values)
    Xs = scaler.transform(X.values)
    pca = PCA(n_components=n_components, random_state=random_state).fit(Xs)

    loadings = pd.DataFrame(
        pca.components_.T, index=feature_names,
        columns=[f"PC{i+1}" for i in range(pca.n_components_)]
    )
    evr = pd.Series(pca.explained_variance_ratio_, index=loadings.columns,
                    name="explained_variance_ratio")

    if verbose:
        print(f"[PCA] Using features ({len(feature_names)}): {feature_names}")
        print(f"[PCA] X shape to PCA: {Xs.shape}, n_components={pca.n_components_}")
    return scaler, pca, loadings, evr, feature_names

# ---------- driver ----------
def analyze_pitchers_with_global_pca(player_ids, year, min_rows=50, n_components=None, feature_cols=None):
    pooled, counts = build_pooled_dataset(player_ids, year, min_rows=min_rows, feature_cols=feature_cols)
    scaler, pca, global_loadings, global_evr, feature_names = fit_global_pca_safe(
        pooled, feature_cols=feature_cols, n_components=n_components, verbose=True
    )

    rows_summary, imp_rows = [], []
    for pid, g in pooled.groupby("pitcher_id"):
        g = _dedupe_cols(g)
        X = g[feature_names].apply(pd.to_numeric, errors="coerce")
        y = g[TARGET_COL]
        mask = X.notnull().all(axis=1) & y.notnull()
        X, y = X.loc[mask].values, y.loc[mask]

        # PDI aligned to PCA features
        g_aligned = g.loc[mask, feature_names + [TARGET_COL]]
        pdi = deception_score(g_aligned, feature_cols=feature_names)

        n_classes = y.nunique()
        baseline = 1.0 / n_classes
        cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
        rf = RandomForestClassifier(n_estimators=200, random_state=42)
        acc = cross_val_score(rf, X, y, cv=cv, scoring="accuracy").mean()
        norm_acc = (acc - baseline) / (1 - baseline) if n_classes > 1 else np.nan

        rf.fit(X, y)
        imp_rows.append(pd.Series(rf.feature_importances_, index=feature_names, name=pid))

        scores = pca.transform(scaler.transform(X))
        pc_means = scores.mean(axis=0)
        pc_stds  = scores.std(axis=0)

        row = {
            "pitcher_id": pid,
            "n_rows": int(len(X)),
            "n_classes": int(n_classes),
            "baseline_acc": baseline,
            "rf_cv_acc": acc,
            "rf_cv_norm_acc": norm_acc,
            "PDI": pdi
        }
        for i in range(pca.n_components_):
            row[f"PC{i+1}_mean"] = pc_means[i]
            row[f"PC{i+1}_std"]  = pc_stds[i]
        rows_summary.append(row)

    summary_df = pd.DataFrame(rows_summary).sort_values("PDI", ascending=True)
    importances_df = pd.DataFrame(imp_rows)

    return {
        "pooled_counts": counts,
        "summary_per_pitcher": summary_df,
        "rf_feature_importances_per_pitcher": importances_df,
        "mean_rf_feature_importance": importances_df.mean(axis=0).sort_values(ascending=False).rename("mean_importance"),
        "global_pca_loadings": global_loadings,
        "global_pca_explained_variance": global_evr,
        "pc1_pc2_by_pitcher": summary_df[["pitcher_id","PC1_mean","PC2_mean","PDI","rf_cv_acc","n_rows"]].copy(),
        "feature_names": feature_names,
        "scaler": scaler,
        "pca": pca
    }

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

def _get_evr(results):
    evr = results.get("global_pca_explained_variance", None)
    if evr is None:
        raise KeyError("results['global_pca_explained_variance'] not found.")
    # Accept Series or 1-col DataFrame
    if isinstance(evr, pd.Series):
        return evr.values
    if isinstance(evr, pd.DataFrame):
        if evr.shape[1] == 1:
            return evr.iloc[:, 0].values
        # if columns are like ['PC1','PC2',...] treat as row of one record
        if evr.shape[0] == 1:
            return evr.values.ravel()
        # last resort: if there is a column literally named 'explained_variance_ratio'
        if 'explained_variance_ratio' in evr.columns:
            return evr['explained_variance_ratio'].values
    raise ValueError("Unrecognized format for global_pca_explained_variance.")

# 1) Scree plot
def plot_pca_scree(results, figsize=(6,4)):
    evr_vals = _get_evr(results)
    xs = np.arange(1, len(evr_vals)+1)
    plt.figure(figsize=figsize)
    plt.plot(xs, evr_vals, marker="o")
    plt.xticks(xs)
    plt.xlabel("Principal Component")
    plt.ylabel("Explained Variance Ratio")
    plt.title("Scree Plot (Global PCA)")
    plt.tight_layout()
    plt.show()
